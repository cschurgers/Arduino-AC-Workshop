/**********************************************************************************
   LED Tutorial

   This tutorial demonstrates how to blink the LED on-off using digitalWrite().
***********************************************************************************/

const int ledPin = 13;                    // LED pin number



// Setup function
void setup() {
  pinMode(ledPin, OUTPUT);                // The digital pin is configured as an output
}


// Main function (executed continuously)
void loop () {                             
  digitalWrite(ledPin, HIGH);             // Turn the LED on by setting the ledPin to HIGH(5V)
  delay(1000);                            // Wait for 1000 ms
  digitalWrite(ledPin, LOW);              // Turn the LED off by setting the ledPin to LOW(0V)
  delay(1000);                            // Wait for 1000 ms


}
