Photocell


 
 
 

 
Photocell 
Read light intensity values with a photocell.

Detailed Discussion

  • Photocell - discussion

For this tutorial, you need to open the "Serial Monitor" as in the SerialForwarder tutorial. In the menu bar of the Arduino IDE, click on Tools, and then click on Serial Monitor.
Resistor code:


 


  


  
/**********************************************************************************
   Photocell Tutorial

   This tutorial demonstrates how to hook up a photocell. A photocell is nothing more
   than a resistor of which the resistance changes with the amount of light falling on it.
   We read an analog value from the circuit through an analog input pin.
***********************************************************************************/

const int sensorPin = A2;                 // Pin for the light sensor (analog pin)
                                          // An analog pin (pin name is A followed by a number) is always an input,
                                          // so don't need to specify in setup() that it is an input
const int ledPin = 9;                     // LED pin
const int buttonPin = 2;                  // Push button pin 

int buttonState;                          // Variable to store state
int sensorReading;


void setup() {
  Serial.begin(9600);                     // Set up the serial connection to your computer

  pinMode(ledPin, OUTPUT);                // The digital pin configured as an output
  pinMode(buttonPin, INPUT);              // The digital pin configured as an input

  digitalWrite(ledPin, HIGH);             // Turn the LED on

}



void loop () {

  sensorReading = analogRead(sensorPin);    // Read from the sensor
  
  Serial.print("Sensor reading = ");        // Print this text in the Serial Monitor
  Serial.println(sensorReading);            // Print the sensor reading, followed by a newline character

  if (sensorReading > 800) {                // You can modify this value so that you get to see a difference
                                            // when you change the light level                           
    digitalWrite(ledPin, HIGH);             // Turn the LED on
  }
  else {
    digitalWrite(ledPin, LOW);              // Turn the LED off
  }

  buttonState = digitalRead(buttonPin);
  if (buttonState == LOW) {                 // Button is pressed             
    Serial.println("Button is pressed");           
  }
         
  delay(50);

}
